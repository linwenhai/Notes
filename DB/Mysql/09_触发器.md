## 触发器



```sql
/* 1 创建测试表*/
mysql> create table test_1(id int,name varchar(10));
mysql> create table test_2(id int,name varchar(10));
```

```sql
/* 2
创建触发器ins_test,new代表刚才更新或者插入的记录
*/
mysql> 
delimiter $$
create trigger ins_test
after insert 
on test_1 for each row 
begin
insert into test_2 values(new.id,new.name);
end $$
delimiter ;
```

```sql
/* 3 测试*/
mysql> insert into test_1 values(1,'lin hai');
mysql> select * from test_1;
mysql> select * from test_2;
```

```sql
/* 4 查看触发器*/
mysql> show triggers \G
```

```sql
/* 5 删除触发器*/
mysql> drop trigger ins_test;
```
